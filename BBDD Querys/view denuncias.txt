CREATE VIEW `V_DENUNCIA` 

AS 

SELECT 

    `E`.`TIPO_ESTADO_PUBLICACION` AS `ESTADO_PUBLICACION`,

    `P`.`ID_USUARIO` AS `USUARIO_DENUNCIADO`,

    `D`.`ID_PUBLICACION` AS `ID_PUBLICACION`,

    `D`.`ID_DENUNCIA` AS `DENUNCIA`

    FROM ((`DENUNCIA` `D` 


    JOIN `PUBLICACION` `P` ON((`P`.`ID_PUBLICACION` = `D`.`ID_PUBLICACION`)))

    JOIN `ESTADO` `E` ON((`E`.`ID_ESTADO` = `P`.`ID_ESTADO`)));
