CREATE TABLE `ROL_PERMISO` (
  `ID_ROL_PERMISO` INT NOT NULL AUTO_INCREMENT,
  `ID_ROL` INT NOT NULL,
  `ID_PERMISO` BIGINT NOT NULL,
  PRIMARY KEY (`ID_ROL_PERMISO`),
  KEY `ROL_PERMISOS_ROLES` (`ID_ROL`),
  KEY `PERMISO_PERMISOS_ROLES` (`ID_PERMISO`),
  CONSTRAINT `PERMISO_PERMISOS_ROLES` FOREIGN KEY (`ID_PERMISO`) REFERENCES `PERMISOS` (`ID_PERMISO`),
  CONSTRAINT `ROL_PERMISOS_ROLES` FOREIGN KEY (`ID_ROL`) REFERENCES `ROL` (`ID_ROL`)
) ENGINE=INNODB DEFAULT CHARSET=UTF8MB3;

CREATE TABLE `ROL` (
  `ID_ROL` INT NOT NULL AUTO_INCREMENT,
  `NOMBRE_ROL` VARCHAR(100) NOT NULL,
  `DESCRIPCION` VARCHAR(100) NOT NULL,
  PRIMARY KEY (`ID_ROL`),
  UNIQUE KEY `ID_ROL` (`ID_ROL`,`NOMBRE_ROL`)
) ENGINE=INNODB DEFAULT CHARSET=UTF8MB3;

CREATE TABLE `USUARIO_ROL` (
  `ID_USUARIO_ROL` INT NOT NULL AUTO_INCREMENT,
  `ID_USUARIO` BIGINT NOT NULL,
  `ID_ROL` INT NOT NULL,
  PRIMARY KEY (`ID_USUARIO_ROL`),
  KEY `ROL_USUARIO_ROL` (`ID_ROL`),
  KEY `USUARIO_USUARIO_ROL` (`ID_USUARIO`),
  CONSTRAINT `ROL_USUARIO_ROL` FOREIGN KEY (`ID_ROL`) REFERENCES `ROL` (`ID_ROL`),
  CONSTRAINT `USUARIO_USUARIO_ROL` FOREIGN KEY (`ID_USUARIO`) REFERENCES `USUARIO` (`ID_USUARIO`)
) ENGINE=INNODB DEFAULT CHARSET=UTF8MB3;
